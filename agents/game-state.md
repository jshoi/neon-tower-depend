## 파일 구조

```
src/
├── App.js                       # 메인 앱, scale 계산, 게임루프
├── index.css                    # 전역 스타일, touch-action
├── config/
│   ├── gameConfig.js            # 캔버스/타워/적/웨이브 수치
│   └── upgradeConfig.js         # 12종 업그레이드 정의
├── game/
│   ├── GameManager.js           # 게임 루프, 상태 관리, 업그레이드
│   ├── WaveSystem.js            # 웨이브 상태머신
│   ├── TowerSystem.js           # 타워 공격/방어 로직
│   ├── EnemySystem.js           # 적 스폰/이동
│   └── CollisionSystem.js       # 충돌 처리
├── components/
│   ├── GameCanvas.jsx           # 배경 그리드, 이펙트 렌더링
│   ├── Tower.jsx                # 육각형 타워, 사거리 원, 체력바
│   ├── Enemy.jsx                # 사각형 적, 둔화 색상
│   ├── HUD.jsx                  # 상단 HUD
│   ├── UpgradePanel.jsx         # 업그레이드 패널
│   ├── GameOver.jsx             # 게임 오버 화면
│   └── CoinAnimation.jsx        # 코인 텍스트 오버레이
└── utils/
    ├── mathUtils.js             # 거리 계산, 스폰 위치
    └── animationUtils.js        # 파티클, 텍스트 애니메이션
```
